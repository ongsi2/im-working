import { useState, useEffect, useCallback } from 'react';
import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
import { useAuth } from './hooks/useAuth';
import { useDashboard } from './hooks/useDashboard';
import { DashboardLayout } from './components/DashboardLayout';
import { MetricsGrid, GrowthChart, ActivityFeed } from './components';
import type { DashboardData, User } from './lib/types';
import './main.css';
const queryClient = new QueryClient({
defaultOptions: { queries: { staleTime: 1000 * 60 * 5 } },
});
interface AppState {
isLoading: boolean;
error: Error | null;
data: DashboardData | null;
}
export default function App(): JSX.Element {
const { user, isAuthenticated, signOut } = useAuth();
const [state, setState] = useState<AppState>({
isLoading: true, error: null, data: null,
});
const fetchDashboard = useCallback(async () => {
try {
setState(s => ({ ...s, isLoading: true }));
const res = await fetch(`/api/dashboard?uid=${user?.id}`);
const json: DashboardData = await res.json();
setState({ isLoading: false, error: null, data: json });
} catch (err) {
setState(s => ({ ...s, isLoading: false, error: err as Error }));
}
}, [user?.id]);
useEffect(() => {
if (isAuthenticated) fetchDashboard();
}, [isAuthenticated, fetchDashboard]);
if (!isAuthenticated) return <LoginPage />;
return (
<QueryClientProvider client={queryClient}>
<DashboardLayout user={user as User} onSignOut={signOut}>
<MetricsGrid data={state.data} loading={state.isLoading} />
<GrowthChart metrics={state.data?.metrics} />
<ActivityFeed items={state.data?.activity} />
</DashboardLayout>
</QueryClientProvider>
);
}
(alias) function fetchDashboard(): Promise<void>
Fetches dashboard data for the authenticated user and updates component state. Debounces on user.id changes.
