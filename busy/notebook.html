# 코호트 베이스 — 가입 주차별 유저 수
conn = connect(user="analytics_ro", account="novafield-prod", role="ANALYST")

query = """
SELECT DATE_TRUNC('week', signup_at) AS cohort_week,
COUNT(DISTINCT user_id) AS new_users
FROM analytics.dim_users
WHERE signup_at >= '2025-07-01'
GROUP BY 1 ORDER BY 1
"""
cohort = pd.read_sql(query, conn)

# 유지율 매트릭스 계산
activity = pd.read_parquet("datasets/events.parquet")
activity["weeks_since_signup"] = (
  (activity.event_week - activity.cohort_week).dt.days // 7
)
retention = (
  activity.groupby(["cohort_week", "weeks_since_signup"])
  .user_id.nunique().unstack().divide(cohort.set_index("cohort_week").new_users, axis=0)
)
retention.round(3)
